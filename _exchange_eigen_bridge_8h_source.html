40  ts3d::A3DMiscCartesianTransformationWrapper d( xform );
41  auto const mirror = (d->m_ucBehaviour & kA3DTransformationMirror) ? -1. : 1.;
42  auto const s = ts3d::getVector( d->m_sScale );
43  auto const o = ts3d::getPosition( d->m_sOrigin );
44  auto const x = ts3d::getVector( d->m_sXVector );
45  auto const y = ts3d::getVector( d->m_sYVector );
46  auto const z = x.cross3( y ) * mirror;
47 
48  ts3d::MatrixType result;
49  for(auto idx = 0u; idx < 4u; idx++) {
50  result( idx, 0 ) = x( idx ) * s( 0 );
51  result( idx, 1 ) = y( idx ) * s( 1 );
52  result( idx, 2 ) = z( idx ) * s( 2 );
53  result( idx, 3 ) = o( idx );
54  }
55

57  }
