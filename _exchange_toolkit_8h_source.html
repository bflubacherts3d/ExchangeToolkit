1295  // This function can be used to obtain a set of type paths where the parent type
1296  // and leaf child type values are both given. Each type path returned is one
1297  // possible way to traverse the Exchange data structures such that the beginning
1298  // entity type and ending entity type are equal to the input parameters.
1299  static TypePathArray getPossibleTypePaths( A3DEEntityType const &parent_type, A3DEEntityType const &child_type ) {
1300 
1301  // check to see if the given owner and child type values are present in the cache
1302  static std::unordered_map< A3DEEntityType, std::unordered_map< A3DEEntityType, TypePathArray >> _cache;
1303  auto const parent_type_it = _cache.find( parent_type );
1304  if(std::end( _cache ) != parent_type_it ) {
1305  auto const child_type_it = parent_type_it->second.find( child_type );
1306  if( std::end( parent_type_it->second ) != child_type_it ) {
1307  return child_type_it->second;
1308  }
1309  }
1310 
1311  // result is not present in the cache, so compute it
1312  TypePathArray type_paths;
1313  auto const possible_parents = getPossibleParents( child_type );
1314  if( std::end( possible_parents ) != possible_parents.find( parent_type ) ) {
1315  // child entity type is an immediate child of parent entity type
1316  TypePath type_path;
1317  type_path.push_back( parent_type );
1318  type_path.push_back( child_type );
1319  type_paths.push_back( type_path );
1320  } else {
1321  // child entity type is not an immediate child of parent entity type
1322  // so we must iterate over the list of all possible parent types
1323  // and for each of those parent types, examine up the ownership chain
1324  for( auto const intermediate_parent_type : possible_parents ) {
1325  if( intermediate_parent_type != child_type ) {
1326  auto const intermediate_type_paths = getPossibleTypePaths( parent_type, intermediate_parent_type );
1327  for( auto const intermediate_type_path : intermediate_type_paths ) {
1328  type_paths.push_back( intermediate_type_path );
1329  type_paths.back().push_back( child_type );
1330  }
1331  }
1332  }
1333  }
1334  // augment the cache
1335 
1336  _cache[parent_type][child_type] = type_paths;
1337  return type_paths;
1338  }
